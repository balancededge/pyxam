\question[6] An amplifier has an open loop gain of $P(s) = K$  and then feedback $C(s) = \beta$ is added as shown in Figure~\ref{sens_fig}.
\begin{parts}
    \part Find the sensitivity of the open loop system shown in Figure~\ref{sens_fig}(a) with respect to $K$.
    \part Find the sensitivity of the closed loop system shown in Figure~\ref{sens_fig}(b) with respect to $K$.
    \part For parts (a) and (b) above what is the sensitivity with $K =10^4$ and $\beta = 0.1$.
\end{parts}

\begin{figure} [ht!]

            \node [output, right of=K] (output) {};
            \draw [draw,->] (input) -- node {$r$} (K);
            \draw [->] (K) -- node [name=y] {$y$}(output);
        \end{tikzpicture}
    }
    \subfloat[Closed Loop Schematic]{
        \begin{tikzpicture}[auto, node distance=2cm,>=latex']
            \node [input, name=input] {};

            \node [block, right of=sum] (system) {P(s) = K};
            \node [block, below of=system] (return) {$C(s) = \beta$};
            \node [output, right of=system] (output) {};
            \draw [draw,->] (input) -- node[pos=0.9] {$+$}  node [near start] {$r$} (sum);

        \end{tikzpicture}
    }
    \caption{Amplifier open and closed loop block diagram \label{sens_fig}}
